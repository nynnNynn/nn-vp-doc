# 046全排列
## code
```javascript
/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var permute = function(nums) {
    let result = [];
    let used = new Array(nums.length).fill(0);
    function backtrace(nums, temp){
        if(temp.length === nums.length){
            result.push([...temp]);
            return;
        }
        for(let i = 0; i < nums.length; i++){
            if(used[i] !== 1){
                temp.push(nums[i]);
                used[i] = 1;
                backtrace(nums, temp);
                temp.pop();
                used[i] = 0;
            } else {
                continue;
            }
        }
    }
    backtrace(nums, []);
    return result;   
};
```
