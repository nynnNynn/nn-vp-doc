var rightSideView = function(root) {
    let result = [];
    function checkRight(root, level){
        if(root === null) {
            return;
        }
        if(result[level] === undefined){
            result[level] = root.val;
        } else {
            if(root.val){
                result[level] = root.val;
            }
            
        }
        checkRight(root.left, (level + 1));
        checkRight(root.right, (level + 1));

    }
    checkRight(root, 0);
    return result;
};
