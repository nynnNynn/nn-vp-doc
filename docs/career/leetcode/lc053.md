# 053最大子数组和
## code
```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxSubArray = function(nums) {
    let dp = new Array(nums.length);
    let max;
    for(let i = 0; i < nums.length; i++){
        if(i === 0){
            dp[i] = nums[i];
            max = dp[i];
        } else {
            dp[i] = Math.max(dp[i-1] + nums[i], nums[i]);
        }
        max = Math.max(max, dp[i]);
    }
    return max;  
};
```
