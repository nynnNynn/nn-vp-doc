# 179最大数
## code
```javascript
/**
 * @param {number[]} nums
 * @return {string}
 */
var largestNumber = function(nums) {
    if(nums.every(item => item === 0)){
        return '0';
    }
    return nums.sort(
        (a,b) => {
            if(Number(String(a) + String(b)) >= Number(String(b) + String(a)) ){
                // ab > ba a在前面
                return -1;
            } else {
                return 1;
            }
        }
    ).map(item => String(item)).reduce((accu, curr) => accu + curr);
};
```
