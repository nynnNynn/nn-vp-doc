# 122两数和
## code
```javascript
/**
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function(prices) {
    let temp_min = Infinity;
    // 局部最小
    let result = 0;
    for(let i = 0; i < prices.length; i++){
        temp_min = Math.min(temp_min, prices[i]);
        if(prices[i] > temp_min){
            result = result + prices[i] - temp_min;
            temp_min = prices[i];
        }
    }
    return result;
};
```
