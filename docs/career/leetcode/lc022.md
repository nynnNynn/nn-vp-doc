# 022括号生成
## code
```javascript
/**
 * @param {number} n
 * @return {string[]}
 */
var generateParenthesis = function(n) {
    // 先左再右，左等于右
    // 0 0 左等于右时候，只能落左，左大于右，可落左可落右，

    let result = [];
    function backtrace(str, left, right, n){
        if(str.length === n * 2){
            result.push(str);
            return;
        }
        for(let i = 0; i < n; i++) {
            if(left === right) {
                str = str + '(';
                left++;
                backtrace(str, left, right, n);
                str = str.slice(0, str.length - 1);
            }
        }

    }
    backtrace('', 0, 0, n);
    return result;
    
};
```
