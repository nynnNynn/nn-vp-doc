/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @return {number}
 */
var findMedianSortedArrays = function(nums1, nums2) {
    let ind1 = 0;

    let ind2 = 0;
    [1,2][3,4]
    0,0, 2

    while((ind1 + ind2) < (nums1.length + nums2.length)/2){
        if(nums1[ind1] <= nums2[ind2]){
            ind1++;
        } else {
            ind2++;
        }
    }
    
};
