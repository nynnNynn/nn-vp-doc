# 021 合并两个有序链表

## code

```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} list1
 * @param {ListNode} list2
 * @return {ListNode}
 */
var mergeTwoLists = function (list1, list2) {
  if (list1 === null) {
    if (list2 === null) {
      return null;
    }
    return list2;
  }
  if (list2 === null) {
    return list1;
  }
  let newHead; 
  let result; 
  if (list1.val <= list2.val) {
    newHead = list1;
    list1 = list1.next;
  } else {
    newHead = list2;
    list2 = list2.next;
  }
  result = newHead;
  while (list1.next !== null || list2.next !== null) {
    if (list1 === null) {
      newHead.next = list2;
      newHead = newHead.next;
      list2 = list2.next;
      continue;
    }
    if (list2 === null) {
      newHead.next = list1;
      newHead = newHead.next;
      list1 = list1.next;
      continue;
    }
    // 都没挂
    if (list1.val <= list2.val) {
      newHead.next = list1;
      newHead = newHead.next;
      list1 = list1.next === null ? null : list1.next;
    } else {
      newHead.next = list2;
      newHead = newHead.next;
      list2 = list2.next === null ? null : list2.next;
    }
  }
  if (list1!== null) {
    newHead.next = list1;

  } 
  if(list2 !== null) {
    newHead.next = list2;
  }
  return result;
};
```
