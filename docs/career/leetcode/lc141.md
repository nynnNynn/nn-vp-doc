# 141链表环
## code
```javascript
/**
 * @param {ListNode} head
 * @return {boolean}
 */
var hasCycle = function(head) {
    if (head === null || head.next === null || head.next.next === null){
        return false;
    }
    let fast = head.next.next;
    let slow = head.next;
    while(fast.next !== null && fast.next.next !== null) {
        if(fast === slow) {
            return true;
        }
        fast = fast.next.next;
        slow = slow.next;
    }
    return false;
};
```
