# 003无重复子串

## code

```javascript
/**
 * @param {string} s
 * @return {number}
 */
var lengthOfLongestSubstring = function (s) {
  let left = 0;
  let right = 0;
  let max = 0;
  let used = {};
  while (right < s.length) {
    if (used[s.charAt(right)] !== undefined) {
      delete used[s.charAt(left)];
      left++;
      max = Math.max(max, right - left);
      continue;
    }
    used[s.charAt(right)] = true;
    right++;
    max = Math.max(max, right - left);
  }
  return max;
};
```
