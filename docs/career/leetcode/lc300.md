# 300最长上升子序列
## code
```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */
var lengthOfLIS = function(nums) {
    let dp = new Array(nums.length).fill(0);
    for(let i = 0; i < nums.length; i++){
        if(nums[i] > nums[i-1]){
            dp[i] = dp[i-1] + 1;
        } else {
            dp[i] = Math.max(dp[i-1], 1);
        }
    }

    return dp[nums.length - 1];
    
};
```
